def is_operator(c):
    if c in ['^', '*', '/', '%', '+', '-']:
        return True
    return False

def is_operand(c):
    if c in '0123456789':
        return True
    return False

def operate(v1, v2, o):
    if o == '*':
        return v1*v2
    elif o == '/':
        return v1/v2
    elif o == '%':
        return v1%v2
    elif o == '+':
        return v1+v2
    elif o == '-':
        return v1-v2
    elif o == '^':
        return pow(v1, v2)
    return 0

def eval_postfix(postfix):
    stack = []

    for c in postfix:
        if is_operand(c):
            stack.append(c)
        if is_operator(c):
            v1 = stack.pop()
            v2 = stack.pop()
            r = operate(int(v1), int(v2), c)
            stack.append(r)

    return stack.pop()

def main():
    print(eval_postfix('23+'))
    print(eval_postfix('234*+'))
    print(eval_postfix('234*+5-'))
    print(eval_postfix('23+4*5-'))
    print(eval_postfix('2345*+*'))
    print(eval_postfix('23+45-*6/'))

if __name__ == '__main__':
    main()
