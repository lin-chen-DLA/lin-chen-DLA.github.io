class Rag(object):
    def __init__(self, l, w):
        self.length = l
        self.width = w

    def get_area(self):
        return self.length * self.width

    def __str__(self):
        s = ''
        s = s+'Length: '+str(self.length)+'\n'
        s = s+'Width: '+str(self.width)+'\n'
        s = s+'Area: '+str(self.get_area())+'\n'

        return s

    def __gt__(self, other):
        return self.get_area() > other.get_area()

def main():
    r1 = Rag(10, 20)
    r2 = Rag(5, 40)

    print(r1, r2)
    print(r1 > r2)

if __name__ == '__main__':
    main()
