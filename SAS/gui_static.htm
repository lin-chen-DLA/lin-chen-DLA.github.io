Static GUI
Workflow
  1. Create a Job Definition and add code as the back-end
  2. Create an associated HTML form as the front-end
  3. Save
  4. Run

HTML form
<!DOCTYPE html>
<html lang="end">
<head>
	<meta charset="UTF-8">
</head>
<script src="/SASJobExecution/resources/sap/ui/thirdparty/jquery.js"></script>

<body>
	<form action="/SASJobExecution/" target="_tab">
		<input type="hidden" name="_program" value="$PROGRAM$"> <! -- specifies job definition to run, $PROGRAM$ is the path of the currently executing job -->
		<input type="hidden" name="_action" value="execute"> <! -- executes the job -->
		<input name="var_sex" value="M">
		<input type="submit" value="Submit">
		</form>
</body>
</html>
Job Definition Code
/* get parameter value from GUI */
%let sex = &var_sex;

proc print data=sashelp.class(obs=3);
	where sex = "&sex";
run;
