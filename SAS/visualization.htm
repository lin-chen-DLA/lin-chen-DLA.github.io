/* Histogram Plot */
ods graphics / reset /* clear any previous graphical attribute settings */
		width=4in height=3in /* set up size */
		imagemap=on; /* turn on interactive mode */

proc sgplot data=sashelp.cars;
	histogram MSRP /;
	yaxis grid;
run;

ods graphics / reset;

    PROC SGPANEL
  • Multiple plots displayed together in a panel based on grouping variables
    • /* Panel Scatter Plots */
ods graphics / reset
		width=4in height=3in
		imagemap=on;

proc sgpanel data=sashelp.cars;
	panelby Type; /* Grouping by the variable 'Type' */
	scatter x=weight y=MSRP;
run;

    PROC UNIVARIATE
  • Create statistical plots like histograms, box plots, and QQ plots
    • /* Scatter Plot */
ods graphics / reset
		width=4in height=3in
		imagemap=on;

proc gplot data=sashelp.cars noprint;
	histogram horsepower /
		normal (mu=est sigma=est color=blue w=2.5)
		barlabel = percent
		midpoints = 70 to 550 by 50;
run;

ods graphics / reset;
    PROC KDE
  • Kernel density estimate plots
    • /* Scatter Plot */
ods graphics / reset
		width=4in height=3in
		imagemap=on;

proc kde data=sashelp.cars;
	univar MSRP / plots=density;
run;
