	echo -n "$i "

#!/bin/bash

a=0

while [ $a -lt 10 ]
do
	echo -n "$a "
	a=`expr $a + 1`
done
echo

while [ $a -gt 0 ]
do
	echo -n "$a "
	let a--
done
echo
