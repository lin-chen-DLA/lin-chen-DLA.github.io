    SQL
    # show databases
select * from sys.m_databases

# show schemas
select * from sys.schemas

# show tables in a schema
select * from sys.m_tables where schema_name = 'schema_name'

# show views in a schema
select * from sys.views where schema_name = 'schema_name'

# query table
select * from "schema_name"."table_name"

# change password
alter user user_name password "user_password"
    Import/Export
    # Import
File -> Import -> SAP HANA Conttent -> Data from Local File -> 
select target system -> select data, select schema, input table name ->
update column names -> Finish

# Export
Right click Tables of a schema -> Export -> search target table -> Next ->
select table format -> setup export location -> Finish
