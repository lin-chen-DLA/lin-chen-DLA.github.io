University Database
Start MySQL
# compose.yml
version: "3.9"  # optional since v1.27.0
services:
  mysqlserver:
    image: mysql:8.0.40
    ports:
      - 3306:3306
    volumes:
      - type: volume
        source: sandbox-home
        target: /var/lib/mysql
    environment:
      - MYSQL_ROOT_PASSWORD=root
  phpmyadmin:
    image: phpmyadmin/phpmyadmin
    ports:
      - 8080:80
    environment:
      - PMA_HOST=mysqlserver
volumes:
  sandbox-home:
			
docker-compose up # start MySQL and phpMyAdmin

#docker-compose down # stop MySQL and phpMyAdmin
			
http://localhost:8080/ #root, root
