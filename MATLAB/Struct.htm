Struct
Single Field, Single Value
% Single Field, Single Value
field = 'array';
value = 1:10;
s = struct(field, value)
s.array
Single Field, Multiple Value
% Single Field, Single Value
field = 'array';
value = {'Name', 1:10}
s = struct(field, value)
s(1).array %access the first value
s(2).array %access the second value
Multiple Fields, Multiple Values

% Single Field, Single Value
field1 = 'f1';
value1 = {'Name', 1:10};
field2 = 'f2';
value2 = 1:4;
s = struct(field1, value1, field2, value2);
s(1)
s(2)
s(1).f1 %'Name'
s(2).f1 %1:10
s(1).f2 %1:4 
s(2).f2 %1:4
