Insertion and Deletion
% Delete
a = magic(3)
a(2, :) = [] % delete the second row
a(:, 2) = [] % delete the second column

% Add
b = magic(3);
b(4, :) = 1:3 % insert a row to the end
b(:, 4) = [1; 2; 3; 4] % insert a column to the end
b = [b; 1:2:8] % insert row
