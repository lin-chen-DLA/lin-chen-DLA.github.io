import pandas as pd

# read data
df = pd.read_csv('data.csv')

# convert string to value
def get_profit(s):
    s = s.strip()
    s = s[1:]
    if s[0] == '(':
        return -float(s[1:-1])
    return float(s[1:])

df['Profit'] = df[' Profit_Per_Package '].apply(get_profit)

# convert delivery type to ordinal index
def get_type(t):
    types = {'Same-Day':0, 'Standard (3-5 Days)':1, 'Next-Day':2}
    return types[t]

df['Delivery_Type'] = df['Delivery_Type'].apply(get_type)

# select columns
df = df[['Delivery_Type', 'Delivery_Distance_Miles', 'Package_Weight_Lbs', 'On_Time_Delivery', 'Profit', 'Customer_Retention']]

from scipy.optimize import minimize
import numpy as np

# define objective function
def objective(x):
    x = np.array(x).reshape(1, -1)  # reshape to 2D input
    y_profit = model_profit.predict(x)[0]
    y_retention = model_retention.predict_proba(x)[0][1]

    y = scaler.transform(np.array([y_profit, y_retention]).reshape(1, -1))

    loss = - (weight_profit * y[0][0] + weight_retention * y[0][1])

    return loss
