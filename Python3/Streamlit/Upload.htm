  • UploadedFile is a subclass of io.BytesIO

    • import streamlit as st
import pandas as pd
import io

uploaded_file = st.file_uploader("Choose a file")
if uploaded_file is not None:

    print(type(uploaded_file)) # UploadedFile

    # file information
    print(uploaded_file.name)
    print(uploaded_file.type)
    print(uploaded_file.size)
    print(uploaded_file.file_id)

    # output
    dataframe = pd.read_csv(uploaded_file) # dataframe
    st.write(dataframe)

    print(uploaded_file.getvalue().decode("utf-8")) # string

    with open("copy.csv", "wb") as f:
        f.write(uploaded_file.getbuffer()) # csv file
