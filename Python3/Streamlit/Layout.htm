import streamlit as st

def callback(option):
    if option == 'Email':
        st.text('lin.chen@ieee.org')
    elif option == 'Home phone':
        st.text('757.123.000')
    else:
        st.text('By appointment')

def main():
    option = st.selectbox(
    "How would you like to be contacted?", ("Email", "Home phone", "Other"))

    st.button("Greet", on_click=callback, args=(option,))

if __name__ == '__main__':
    main()
Component Layout

import streamlit as st

def get_content(option):
    if option == 'Email':
        st.text('lin.chen@ieee.org')
    else:
        st.text('By appointment')

def get_option(display):
    if display:
        option = st.selectbox(
        "How would you like to be contacted?", ("Email", "Phone"))
        get_content(option)

def main():
    display = st.checkbox("Display")
    get_option(display)

if __name__ == '__main__':
    main()

Column Layout

import streamlit as st

st.set_page_config(
    page_title="My Streamlit App",  # Title of the browser tab
    page_icon=":us:",  # Icon for the browser tab
    layout="wide",  # Layout can be "centered" or "wide"
    initial_sidebar_state="expanded",  # Sidebar state can be "auto", "expanded", "collapsed"
)

def main():
    # sidebar
    st.sidebar.write('Element 1 in sidebar')
    st.sidebar.write('Element 2 in sidebar')

    # main window
    st.write('Element 1 in main window')
    st.write('Element 2 in main window')

if __name__ == '__main__':
    main()
        

import streamlit as st

def main():
    # row 1
    col1, col2 = st.columns([2, 1])
    with col1:
        st.write('Row 1: Column 1')
    with col2:
        st.write('Row 1: Column 2')

    # row 2
    col3, col4, col5 = st.columns([2, 4, 3])
    with col3:
        st.write('Row 2: Column 1')
    with col4:
        st.write('Row 2: Column 2')
    with col5:
        st.write('Row 2: Column 3')

if __name__ == '__main__':
    main()
