Write YAML File
import yaml

dictionary = {'Name': 'Lin', 'Age': 41, 'Children': ['Hannah', 'Hadley'], 'Lobbies': {'Swim':'Normal', 'Running':'Professional'}}

with open('temp.yaml', 'w') as file:
    yaml.dump(dictionary, file)
