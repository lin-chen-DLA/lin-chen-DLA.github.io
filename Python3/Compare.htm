Compare Objects
class MyAge:
    def __init__(self, n):
        self.n = n

a1 = MyAge(42)
a2 = MyAge(42)

a1 == a2 # False
        
class MyAge:
    def __init__(self, n):
        self.n = n
        
    def __eq__(self, other):
        if not isinstance(other, MyAge):
            return NotImplemented
        return self.n == other.n

a1 = MyAge(42)
a2 = MyAge(42)

a1 == a2 # True
