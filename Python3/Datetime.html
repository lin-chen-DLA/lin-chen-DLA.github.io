    timedelta
  • Create timedelta
    • from datetime import timedelta
from datetime import date
from datetime import datetime

# create timedelta
td = timedelta(days = 2, hours = 2)

d1 = date.fromisoformat('2019-12-04')
d2 = date.fromisoformat('2023-12-04')
td = d2 - d1 # datetime.timedelta
td

d1 = datetime.fromisoformat('2011-11-04T00:05:23')
d2 = datetime.now() # 2021-02-06 00:09:26.719018
td = d2 - d1 # datetime.timedelta
  • Arithmetic
    • td1 = timedelta(days = 2, hours = 2) # datetime.timedelta(days=2, seconds=7200)
td2 = timedelta(days = 2, hours = 2, minutes = 2) # datetime.timedelta(days=2, seconds=7320)

td2 - td1 # datetime.timedelta(seconds=120)
td2 + td2 # datetime.timedelta(days=4, seconds=14640)
td1*2 # datetime.timedelta(days=4, seconds=14400)
td1/2 # datetime.timedelta(days=1, seconds=3600)
td1/td2 # float
td1//4 # datetime.timedelta(days=1, seconds=3600)
td2%td1 # datetime.timedelta(seconds=120)

d1 = date.fromisoformat('2019-12-04')
d1 + td1 # datetime.date(2019, 12, 6)

d1 = datetime.fromisoformat('2011-11-04T00:05:23')
d1 + td1 # datetime.datetime(2011, 11, 6, 2, 5, 23)
  • Access
    • td = timedelta(days = 2, hours = 2)

td.min, td.max, td.resolution
td.total_seconds() # 180000.0
  • Comparison
    • td1 = timedelta(days = 2, hours = 2) # datetime.timedelta(days=2, seconds=7200)
td2 = timedelta(days = 2, hours = 2, minutes = 2) # datetime.timedelta(days=2, seconds=7320)

td1 < td2 # True
