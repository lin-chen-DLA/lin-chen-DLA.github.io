While
#!/usr/bin/python
 
n = 10;
while n > 0:
    print(n)
    n = n-1;
else:
    print("End of Loop ...")
For
#!/usr/bin/python
 
l = [1, 2, 3, 4];
 
#list elements and id
for i in l:
    print(i, id(i))
 
for i in l:
    i = i*10; #i is immutable
    print(i, id(i))
 
for index, ele in enumerate(l):
    l[index] = l[index]*10;
    print(index, l[index], id(l[index]), ele, id(ele))
