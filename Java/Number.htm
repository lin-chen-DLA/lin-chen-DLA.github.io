Primitives and Objects
public class Num
{
	public static void main(String args[])
	{
		int i = 10;
		Integer i2;

		// primitve int to Integer
		i2 = Integer.valueOf(i);
		System.out.println(i2.getClass().getName());

		// Integer to int
		i = i2.intValue();
		System.out.println(i);

		//autoboxing
		i2 = i;
		System.out.println(i2);

		//unboxing
		i = i2;
		System.out.println(i);
	}
}
