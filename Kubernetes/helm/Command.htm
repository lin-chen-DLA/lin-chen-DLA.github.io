Intro
Repository
# search https://artifacthub.io/
# helm search hub [chart]
helm search hub wordpress

# add repository
# helm repo add [repository_name] [url]
helm repo add bitnami https://charts.bitnami.com/bitnami

# search a repository
helm search repo [repository_name]
helm search repo bitnami
Install
# install a package with a specific name
# helm install [release_name] [chart_name]
helm install happy-panda bitnami/wordpress

# install a package and generate a name
# helm install [chart_name] --generate-name
helm install bitnami/mysql --generate-name

# list releases
helm list

# uninstall a release
# helm uninstall [release_name]
helm uninstall mysql-1612624192
