Ingress

Create Ingress
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: minimal-ingress
  annotations:
    nginx.ingress.kubernetes.io/rewrite-target: /
spec:
  ingressClassName: nginx-example
  rules:
  - http:
      paths:
      - path: /testpath
        pathType: Prefix
        backend:
          service:
            name: test
            port:
              number: 80
Ingress Controller in Minikube
# start the k8s Nginx implementation of Ingress Controller
# create ingress-nginx-controller in ingress-nginx namespace
minikube addons enable ingress
        
# start k8s dashboard
# create service "kubernetes-dashboard" in namespace "kubernetes-dashboard"
minikube dashboard
        
# ingress.yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: dashboard-ingress
  namespace: kubernetes-dashboard
spec:
  rules:
  - host: dashboard.com
    http:
      paths:
      - path: /index
        pathType: Prefix
        backend:
          service:
            name: kubernetes-dashboard # service name
            port:
              number: 80 # service port
        
# create ingress
kubectl apply -f ingress.yaml

# get ip and port of ingress
kubectl get ingress -n kubernetes-dashboard
