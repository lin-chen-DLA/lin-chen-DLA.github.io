    Frozenset
  • Frozenset type is immutable
    • #!/usr/bin/python

s = frozenset(['a', 'a', 'b', 'c', 'd']); #['a', 'b', 'c', 'd']
s2 = frozenset(['a', 'd', 'e', 'f']);

print s|s2; #union
print s&s2; #intersection
print s-s2; #difference

if s.issuperset(frozenset(['b'])):
    print '[a, b, c, d] is the super set of [b]'
