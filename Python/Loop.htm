For
#!/usr/bin/python

l = [1, 2, 3, 4];

#list elements and id
for i in l:
    print i, id(i);

for i in l:
    i = i*10; #i is immutable
    print i, id(i);

print l;#[1, 2, 3, 4]

for index, ele in enumerate(l):
    l[index] = l[index]*10;
    print index, l[index], id(l[index]), ele, id(ele);

print l;#[10, 20, 30, 40]
