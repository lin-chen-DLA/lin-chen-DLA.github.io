Struct
  • Conversion between Python values and C structs represented as Python strings
  • it is typically more efficient to do it once when creating a Struct instance and call methods on the instance instead of using the module-level functions
#!/usr/bin/python

import struct
import binascii

# Create Struct instance
s = struct.Struct('hhl');

# Pack information
packed = s.pack(1, 2, 3);#pack data into a string
print s.format
print s.size
print binascii.hexlify(packed)

# Unpack information
unpacked = s.unpack(packed);
print unpacked, type(unpacked);
