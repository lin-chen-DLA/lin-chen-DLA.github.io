Podman Demo
Dockerfile
FROM node:8
WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
EXPOSE 3333
CMD ["node", "app.js"]

Build Image
podman build -t podmandemo .
Deploy Container
podman run -d -p 3333:3333 podmandemo
Access App
http://localhost:3333

Build Image
podman build -t lchenlangley/podmandemo .
Push Image
# login docker hub
podman login docker.io

# push image to docker hub
podman push lchenlangley/podmandemo
Deploy Container
podman run -d -p 3333:3333 lchenlangley/podmandemo
Access App
http://localhost:3333
