Install

    Doc String
  • @file, present in the file header
  • @param, parameters
  • @page, Markdown page name
  • @mainpage, main Markdown page for the project
  • @tableofcontents, generate table of content for the Markdown page
    • /// @file hello.h
#ifndef HELLO_H
#define HELLO_H
 
/** Return a string "Hello World\n"
 * @param [out] a string
 */
char * hello();
 
#endif
			
    /// @file util.h
#ifndef UTIL_H
#define UTIL_H
 
/** Return a double value of an integer
 * @param [in] n an integer number
 * @param [out] an integer number
 */
int doubleNum(int n);
 
#endif
			
    /// @file main.c
#include 
#include "hello.h"
#include "util.h"
 
/**
 * Main entry point of the program.
 */
int main()
{
		printf("%s\n", hello());
		printf("%d\n", doubleNum(10));
			 
		return 0;
}
			
    /// @file hello.c
#include "hello.h"

char * hello()
{
	return "Hello World\n";
}
			
    /// @file util.c
#include "util.h"
 
int doubleNum(int n)
{
		return 2*n;
}
    Generate Documents
  • Doxyfile
    • INPUT, specify the files and/or directories that contain documented source files
      • INPUT = "../src" "../include"
    • FILE_PATTERNS, specify the file types that will be searched
    • RECURSIVE, recrusively search the specified directories
      • RECURSIVE = YES
    • cd doc
doxygen -g // generate "Doxyfile" which is a configuration file
// edit Doxyfile
doxygen // generate documentation
