array
  • Arrays are fixed-size sequence container
#include <iostream>
#include <array>

template <class T>
void display(T it, T end)
{
	for(; it != end; ++it)
		std::cout<<*it<<" ";
	std::cout<<std::endl;
}

int main(int argc, char *argv[])
{
	std::array<int , 10> a;

	//access elements
	for(int i = 0; i < 10; i++)
		a[i] = i;

	std::cout<<a.begin()[1]<<std::endl;

	auto it = a.begin();
	std::advance(it, 2);
	std::cout<<*it<<std::endl;

	display(a.begin(), a.end());

	//size
	std::cout<<a.size()<<std::endl;

	return 0;
}
