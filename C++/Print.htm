setprecision
#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    double a = 3.1415926;

    cout<<setprecision(4)<<a<<"|"<<endl; // 3.142
    cout<<setprecision(3)<<a<<"|"<<endl; // 3.14
    cout<<setprecision(2)<<a<<"|"<<endl; // 3.1

    return 0;
}

#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int a = 100;

    cout<<oct<<a<<endl; // 144

    cout<<showbase<<hex<<a<<endl; // 0x64

    return 0;
}
