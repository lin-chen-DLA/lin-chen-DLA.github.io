Jenkins 
# 1. Pull an Image
# host
docker pull jenkins:2.60.3
		
# 2. Run the Image with bind mount
# host
docker run -p 8080:8080 -p 50000:50000 -v /your/local/directory:/var/jenkins_home jenkins:2.60.3
		
# 3. Check the Initial Password
# host
docker exec -it [containerID/containerName] bash
# container
cat /var/jenkins_home/secrets/initialAdminPassword
		
# 4. Access Jenkins
# host
http://localhost:8080/
